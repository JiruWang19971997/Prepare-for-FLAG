### 395. Longest Substring with At Least K Repeating Characters
- [ ] 重刷
```python
class Solution:
    def longestSubstring(self, s: str, k: int) -> int:
        if k == 0:
            return len(s)
        max_len = 0
        for uniq in range(1, len(set(s)) + 1):
            print('unique', uniq)
            max_len = max(max_len, self.get_longest(s, k, uniq))
        return max_len
                    
    def get_longest(self,s, k, unique = 1):
        
        max_long = 0
        left, right = 0, 0
        count = {}
        uniq_letter = 0
        valid_letter = 0
        while right < len(s):
            # print('right before', left, right)
            # 保证set(string)长度
            count[s[right]] = count.get(s[right], 0) + 1
            
            if count[s[right]] == 1:
                uniq_letter += 1
            # 记录string中每个字符出现次数
            if count[s[right]] == k:
                valid_letter += 1

            if valid_letter == unique and unique == uniq_letter: # 注意两个条件都要满足
                max_long = max(max_long, right - left + 1)
            while uniq_letter > unique and left < right:
                count[s[left]] -= 1
                # 更新set(string)长度
                if count[s[left]] == 0:
                    uniq_letter -= 1
                
                if count[s[left]] == k - 1:
                    valid_letter -= 1
                left += 1

            right += 1
        return max_long
```


# 括号 加减法 stack
### 224. Basic Calculator
```python
import re
class Solution:
    def calculate(self, s: str) -> int:
        stack = []
        n = 0
        num = 0
        s = s.replace(' ', '')
        for i in range(len(s) - 1, -1, -1):
            c = s[i]
            # 数字
            if len(re.findall('[0-9]', c)) > 0:
                digit = int(c)
                num += (10 ** n) * digit
                n += 1
            else:
                if c == '-':
                    stack.append((-1) * num)
                    n = 0
                    num = 0
                elif c == '+':
                    stack.append(num)
                    n = 0
                    num = 0
                elif c == '(':
                    if n != 0:
                        stack.append(num)
                    num = self.compute(stack)
                    n = 1
                    stack.pop()
                else:
                    stack.append(c)
                
        if n:
            stack.append(num)
        return self.compute(stack)
        
    def compute(self, stack):
        res = 0
        while stack and stack[-1] !=  ')':
            res += stack.pop()
        return res
```

### 227.Basic Calculator II
正向遍历
```python
class Solution:
    def calculate(self, s: str) -> int:
        s = s.replace(' ', '')
        s += '+'
        if s[0] == '-':
            op = '-'
            s = s[1:]
        else:
            op = '+'
        n = 0
        num = 0
        stack = []
        for i in range(len(s)):
            c = s[i]
            
            if c.isdigit():
                num = num * 10 + int(c)
                
                continue
            
            if op == '+':
                stack.append(num)
            elif op == '-':
                stack.append((-1) * num)
            elif op == '*':
                stack.append(stack.pop() * num)
            else:
                tmp = stack.pop()
                if (tmp // num < 0) and (tmp % num) != 0:
                    stack.append(tmp // num + 1)
                else:
                    stack.append(tmp // num)
            op = c
            num = 0
        return sum(stack)
```
