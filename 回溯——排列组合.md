# 全排列
#### nums原始数组在迭代中是变化的
46.无重复
```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        #     1([1,2,3])              2   3  
        # 2,       3 ([2, 3])
        # 3([3])， 2 ([2])
        res = []
        perms = []
        n = len(nums)
        def dfs():
            # base case
            if len(perms) == n:
                res.append(perms[:])
                return 
            for idx, j in enumerate(nums):
                # [1, 2, 3]
                perms.append(j)
                nums.pop(idx)
                # [2, 3]
                dfs()
                perms.pop()
                nums.insert(idx, j)
        dfs()
        return res
```
47.有重复
```python
class Solution:
    def permuteUnique(self, nums: List[int]) -> List[List[int]]:
        # 1,   2
        # 1,2  1
        # 2,1  2
        count = {}
        for n in nums:
            count[n] = count.get(n, 0) + 1
        perm = []
        res = []
        def dfs():
            if len(perm) == len(nums):
                res.append(perm[:])
                return 
            for num in count:
                if count[num] > 0:
                    perm.append(num)
                    count[num] -= 1
                    dfs()
                    perm.pop()
                    count[num] += 1
        dfs()
        return res
```
